! ========================================================================
! CONFIGURACIÓN COMPLETA DESDE CERO: GRE sobre IPsec (IKEv1)
! Incluye ISP, PEAR-A, PEAR-B, SWITCHES y PCs
! ========================================================================

! 1. CONFIGURACIÓN: ROUTER ISP
! ========================================================================
configure terminal
hostname ISP
interface Ethernet0/0
 ip address 200.13.67.1 255.255.255.252
 no shutdown
interface Ethernet0/1
 ip address 200.13.67.5 255.255.255.252
 no shutdown
end
write

! 2. CONFIGURACIÓN: ROUTER PEAR-A
! ========================================================================
configure terminal
hostname PEAR-A
! ACL para cifrar el tráfico GRE
access-list 100 permit gre host 200.13.67.2 host 200.13.67.6
! Fase 1: IKEv1
crypto isakmp policy 10
 encryption aes 256
 hash sha256
 authentication pre-share
 group 14
crypto isakmp key cisco123 address 200.13.67.6
! Fase 2: IPsec Transform Set
crypto ipsec transform-set TSET_GRE esp-aes 256 esp-sha256-hmac
 mode transport
! Crypto Map
crypto map CMAP_GRE 10 ipsec-isakmp
 set peer 200.13.67.6
 set transform-set TSET_GRE
 match address 100
! Interfaces
interface Ethernet0/0
 ip address 200.13.67.2 255.255.255.252
 crypto map CMAP_GRE
 no shutdown
interface Ethernet0/1
 ip address 10.13.67.1 255.255.255.128
 no shutdown
! Túnel GRE
interface Tunnel0
 ip address 172.16.0.1 255.255.255.252
 tunnel source Ethernet0/0
 tunnel destination 200.13.67.6
 tunnel mode gre ip
! Rutas
ip route 0.0.0.0 0.0.0.0 200.13.67.1
ip route 10.13.67.128 255.255.255.128 172.16.0.2
end
write

! 4. CONFIGURACIÓN: SWITCH 1
! ========================================================================
configure terminal
hostname Switch1
vlan 10
 name LAN_PEAR_A
interface range Ethernet0/0 - 1
 switchport mode access
 switchport access vlan 10
 spanning-tree portfast
end
write
