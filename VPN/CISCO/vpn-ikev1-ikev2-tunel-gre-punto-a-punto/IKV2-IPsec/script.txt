! 1. CONFIGURACIÓN: ROUTER ISP
! ========================================================================
configure terminal
hostname ISP
interface Ethernet0/0
 ip address 200.13.67.1 255.255.255.252
 no shutdown
interface Ethernet0/1
 ip address 200.13.67.5 255.255.255.252
 no shutdown
end
write

! 2. CONFIGURACIÓN: ROUTER PEAR-A (200.13.67.2)
! ========================================================================
configure terminal
hostname PEAR-A
access-list 100 permit gre host 200.13.67.2 host 200.13.67.6
crypto ikev2 proposal PROP_IKEv2
 encryption aes-cbc-256
 integrity sha256
 group 14
crypto ikev2 policy POL_IKEv2
 proposal PROP_IKEv2
crypto ikev2 keyring KEYRING_IKEv2
 peer PEER_B
  address 200.13.67.6
  pre-shared-key cisco123
crypto ikev2 profile PROF_IKEv2
 match identity remote address 200.13.67.6 255.255.255.255
 local identity address 200.13.67.2
 authentication remote pre-share
 authentication local pre-share
 keyring local KEYRING_IKEv2
crypto ipsec transform-set TSET_GRE esp-aes 256 esp-sha256-hmac
 mode transport
crypto map CMAP_GRE 10 ipsec-isakmp
 set peer 200.13.67.6
 set transform-set TSET_GRE
 set ikev2-profile PROF_IKEv2
 match address 100
interface Ethernet0/0
 ip address 200.13.67.2 255.255.255.252
 crypto map CMAP_GRE
 no shutdown
interface Ethernet0/1
 ip address 10.13.67.1 255.255.255.128
 no shutdown
interface Tunnel0
 ip address 172.13.67.1 255.255.255.252
 tunnel source Ethernet0/0
 tunnel destination 200.13.67.6
 tunnel mode gre ip
ip route 0.0.0.0 0.0.0.0 200.13.67.1
ip route 10.13.67.128 255.255.255.128 172.13.67.2
end
write
